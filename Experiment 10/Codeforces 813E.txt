#include<bits/stdc++.h>
using namespace std;
int a,b,c,n[100010],o[100010],p[100010],t;
vector<int> m[100010];
struct node{
	int x,y,z;
};
node q[3200010];
void f(int &w,int x,int y,int z){
	q[++t]=q[w];
	q[t].z++;
	w=t;
	if(x==y){
		return;
	}
	int mid=(x+y)/2;
	if(z<=mid){
		f(q[w].x,x,mid,z);
	}
	else{
		f(q[w].y,mid+1,y,z);
	}
}
int g(int v,int w,int x,int y,int z){
	if(y<=z){
		return q[w].z-q[v].z;
	}
	int mid=(x+y)/2;
	int s=g(q[v].x,q[w].x,x,mid,z);
	if(z>mid){
		s+=g(q[v].y,q[w].y,mid+1,y,z);
	}
	return s;
}
int main(){
	cin>>a>>b;
	for(int i=1;i<=a;i++){
		cin>>n[i];
		m[n[i]].push_back(i); 
		if(m[n[i]].size()>=b+1){
			o[i]=m[n[i]][m[n[i]].size()-b-1];
		}
		o[i]++;
		p[i]=p[i-1];
		f(p[i],1,a+1,o[i]);
	}
	cin>>c;
	int z=0;
	while(c--){
		int x,y;
		cin>>x>>y;
		x+=z;
		y+=z;
		x%=a;
		y%=a;
		x++;
		y++;
		if(x>y){
			swap(x,y);
		}
		z=g(p[x-1],p[y],1,a+1,x);
		cout<<z<<endl;
	}
	return 0;
}
