#include<bits/stdc++.h>
using namespace std;
long long answer[200005];
map<long long,int>m;
int main(){
    long long n,l,r;
	cin>>n;
    for(int i=0;i<n;i++){
        cin>>l>>r;
		m[l]++;
        m[r+1]--;
    }
    long long int j=0,k=0;
    for(auto&x:m){
        answer[k]+=(x.first-j);
    	j=x.first;
    	k+=x.second;
    }
    for(int i=1;i<=n;i++){
        cout<<answer[i]<<" ";
    }
    return 0;
}
